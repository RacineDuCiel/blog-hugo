{{- $u := urls.Parse .Destination -}} {{- $src := $u.String -}} {{- $alt := .Text -}} {{- $title := .Title -}} {{- if $u.IsAbs -}} {{/* Remote image: just render standard tag */}} {{- else -}} {{/* Local image: try to find in Page Bundle */}} {{- $img := .Page.Resources.GetMatch $src -}} {{- if $img -}} {{/* Image found in bundle: Resize if too large, convert to WebP */}} {{- $processed := $img -}} {{- if gt $img.Width 800 -}} {{- $src400 := $img.Resize "400x webp q90" -}} {{- $src800 := $img.Resize "800x webp q90" -}} {{- else -}} {{- $processed = $img.Resize (printf "%dx%d webp q90" $img.Width $img.Height) -}} {{- end -}} {{- else -}} {{/* Image not found in bundle (maybe in static?): Fallback */}} {{- end -}} {{- end -}}
